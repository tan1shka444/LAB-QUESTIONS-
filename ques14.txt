#include <stdio.h>
#include <string.h>

int main() {
    char str[100];
    printf("Enter a string: ");
    gets(str);

    int count = 0;
    for (int i = 0; str[i] != '\0'; i++) {
        if ((str[i] == ' ' && str[i+1] != ' ') || (i == 0 && str[i] != ' '))
            count++;
    }

    printf("Number of words: %d\n", count);
    return 0;
}
