#include <stdio.h>

int main() {
 int n1, n2, i, j;
 int arr1[100],arr2 [100];
 printf("Enter size of first array: ");
 scanf("%d", &n1);
 printf("Enter %d elements of first array:\n", n1);
 for (i = 0; i < n1; i++) {
 scanf("%d", &arr1[i]); 
}
printf("Enter size of second array: ");
 scanf("%d", &n2);
 printf("Enter %d elements of second array:\n", n2);
 for (i = 0; i < n2; i++) {
 scanf("%d", &arr2[i]);
}
printf("Common elements are:\n");
 for (i = 0; i < n1; i++) {
     for (j = 0; j < n2; j++) {
     if (arr1[i] == arr2[j]) {
         printf("%d", arr1[i]);
         break; // to avoid printing same element multiple times
     }
 }
 }
    return 0;
}
