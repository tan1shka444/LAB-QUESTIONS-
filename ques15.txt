#include <stdio.h>

int main() {
    int arr[] = {1, 2, 3, 4};
    int n = 4;
    int left[n], right[n], prod[n];

    left[0] = 1;
    for (int i = 1; i < n; i++)
        left[i] = left[i-1] * arr[i-1];

    right[n-1] = 1;
    for (int i = n-2; i >= 0; i--)
        right[i] = right[i+1] * arr[i+1];

    for (int i = 0; i < n; i++)
        prod[i] = left[i] * right[i];

    printf("Output: ");
    for (int i = 0; i < n; i++)
        printf("%d ", prod[i]);
    return 0;
}
